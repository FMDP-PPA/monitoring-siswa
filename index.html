function hitungKalkulasiProgress(rowsDARSiswa, rowsTARSiswa) { const hariIni = new Date(); const tahunSekarang = hariIni.getFullYear(); const bulanSekarang = hariIni.getMonth(); const tglSekarang = hariIni.getDate(); const namaBulan = ["Januari", "Februari", "Maret", "April", "Mei", "Juni", "Juli", "Agustus", "September", "Oktober", "November", "Desember"]; document.getElementById('title-kalender').innerText = `📆 KELENGKAPAN DAR HARIAN (${namaBulan[bulanSekarang].toUpperCase()} ${tahunSekarang})`; const jumlahHariBulanIni = new Date(tahunSekarang, bulanSekarang + 1, 0).getDate(); // Buat array pencatat jam per tanggal let mapJamPerHari = {}; for(let d = 1; d <= jumlahHariBulanIni; d++) { mapJamPerHari[d] = 0; } let totalJamHariIni = 0; let totalJamBulanIni = 0; let totalTarBulanIni = 0; // FUNGSI UTAMA PARSING TANGGAL - DIPERKUAT ANTI-1899 function parsingTanggalGSheets(tglRaw) { if (!tglRaw) return null; // 1. Jika input sudah berupa objek Date bawaan JavaScript/Google Sheets if (tglRaw instanceof Date && !isNaN(tglRaw)) { return { hari: tglRaw.getDate(), bulan: tglRaw.getMonth(), tahun: tglRaw.getFullYear() }; } let tglStr = tglRaw.toString().trim(); if (tglStr === "" || tglStr.toLowerCase().includes("null") || tglStr.toLowerCase().includes("undefined")) return null; // 2. Jika format string ISO standar dari server (contoh: "2026-07-04T07:00:00.000Z" atau "2026-07-04") if (tglStr.includes('-')) { const bagianKiri = tglStr.split('T')[0]; const p = bagianKiri.split('-'); if (p.length === 3) { // Urutan pasti YYYY-MM-DD let thn = parseInt(p[0]); let bln = parseInt(p[1]) - 1; let tgl = parseInt(p[2]); if (thn > 1900) return { hari: tgl, bulan: bln, tahun: thn }; } } // 3. Jika format string garis miring standar lokal (contoh: "04/07/2026" atau "7/4/2026") if (tglStr.includes('/')) { const p = tglStr.split('/'); if (p.length === 3) { // Cek apakah angka pertama Tahun (YYYY/MM/DD) atau Hari (DD/MM/YYYY) if (p[0].length === 4) { return { hari: parseInt(p[2]), bulan: parseInt(p[1]) - 1, tahun: parseInt(p[0]) }; } else { return { hari: parseInt(p[0]), bulan: parseInt(p[1]) - 1, tahun: parseInt(p[2]) }; } } } // 4. Alternatif terakhir jika dikirim sebagai string timestamp atau format terbaca browser let ujiDate = new Date(tglStr); if (!isNaN(ujiDate) && ujiDate.getFullYear() > 1900) { return { hari: ujiDate.getDate(), bulan: ujiDate.getMonth(), tahun: ujiDate.getFullYear() }; } return null; } // Hitung jam dari data DAR milik siswa terpilih rowsDARSiswa.forEach(row => { const keyDurasi = Object.keys(row).find(k => k.toLowerCase().includes('durasi') || k.toLowerCase().includes('jam')); const keyTanggal = Object.keys(row).find(k => k.toLowerCase().includes('tanggal')); let durasi = parseFloat(row[keyDurasi]) || 0; let tglRaw = row[keyTanggal]; const dataTgl = parsingTanggalGSheets(tglRaw); if (dataTgl) { // Hanya hitung jika masuk bulan dan tahun yang berjalan saat ini if (dataTgl.bulan === bulanSekarang && dataTgl.tahun === tahunSekarang) { totalJamBulanIni += durasi; // Masukkan durasi jam ke tanggal yang sesuai di kalender if(mapJamPerHari[dataTgl.hari] !== undefined) { mapJamPerHari[dataTgl.hari] += durasi; } // Jika hari ini, tambahkan ke target harian if (dataTgl.hari === tglSekarang) { totalJamHariIni += durasi; } } } }); // Hitung jumlah TAR milik siswa terpilih rowsTARSiswa.forEach(row => { const keyTanggal = Object.keys(row).find(k => k.toLowerCase().includes('tanggal') || k.toLowerCase().includes('analisis')); let tglRaw = row[keyTanggal]; const dataTgl = parsingTanggalGSheets(tglRaw); if (dataTgl) { if (dataTgl.bulan === bulanSekarang && dataTgl.tahun === tahunSekarang) { totalTarBulanIni++; } } }); const targetDarDaily = 10; const targetDarMonthly = 10 * jumlahHariBulanIni; const targetTarMonthly = 2; let pDarDaily = Math.min(Math.round((totalJamHariIni / targetDarDaily) * 100), 100); let pDarMonthly = Math.min(Math.round((totalJamBulanIni / targetDarMonthly) * 100), 100); let pTarMonthly = Math.min(Math.round((totalTarBulanIni / targetTarMonthly) * 100), 100); // Perbarui Tampilan Lingkaran updateCircularProgress(pDarDaily, pDarMonthly, pTarMonthly, totalJamHariIni, totalJamBulanIni, targetDarMonthly, totalTarBulanIni); // Render Kalender Absensi DAR di bawahnya renderKalenderDAR(jumlahHariBulanIni, mapJamPerHari); }
