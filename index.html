function hitungKalkulasiProgress(rowsDAR, rowsTAR) { const hariIni = new Date(); const tahunSekarang = hariIni.getFullYear(); const bulanSekarang = hariIni.getMonth(); // 0 = Januari, 6 = Juli, dst. const tglSekarang = hariIni.getDate(); // Hitung total hari di bulan berjalan saat ini const jumlahHariBulanIni = new Date(tahunSekarang, bulanSekarang + 1, 0).getDate(); let totalJamHariIni = 0; let totalJamBulanIni = 0; let totalTarBulanIni = 0; // Fungsi pembantu untuk mengurai tanggal dari spreadsheet secara fleksibel function parsingTanggalGSheets(tglStr) { if (!tglStr) return null; // Jika formatnya sudah berupa objek Date atau string ISO (mengandung 'T' atau '-') if (tglStr.includes('-')) { const p = tglStr.split('T')[0].split('-'); if (p.length === 3) { return { hari: parseInt(p[2]), bulan: parseInt(p[1]) - 1, // jadikan 0-11 tahun: parseInt(p[0]) }; } } // Jika formatnya slash/garis miring (DD/MM/YYYY) if (tglStr.includes('/')) { const p = tglStr.split('/'); if (p.length === 3) { return { hari: parseInt(p[0]), bulan: parseInt(p[1]) - 1, // jadikan 0-11 tahun: parseInt(p[2]) }; } } return null; } // 1. KALKULASI DATA DAR rowsDAR.forEach(row => { const keyDurasi = Object.keys(row).find(k => k.toLowerCase().includes('durasi') || k.toLowerCase().includes('jam')); const keyTanggal = Object.keys(row).find(k => k.toLowerCase().includes('tanggal')); let durasi = parseFloat(row[keyDurasi]) || 0; let tglRaw = row[keyTanggal] ? row[keyTanggal].toString().trim() : ''; const dataTgl = parsingTanggalGSheets(tglRaw); if (dataTgl) { // Cek apakah masuk bulan ini & tahun ini if (dataTgl.bulan === bulanSekarang && dataTgl.tahun === tahunSekarang) { totalJamBulanIni += durasi; // Cek apakah tepat hari ini if (dataTgl.hari === tglSekarang) { totalJamHariIni += durasi; } } } }); // 2. KALKULASI DATA TAR BULAN INI rowsTAR.forEach(row => { const keyTanggal = Object.keys(row).find(k => k.toLowerCase().includes('tanggal') || k.toLowerCase().includes('analisis')); let tglRaw = row[keyTanggal] ? row[keyTanggal].toString().trim() : ''; const dataTgl = parsingTanggalGSheets(tglRaw); if (dataTgl) { if (dataTgl.bulan === bulanSekarang && dataTgl.tahun === tahunSekarang) { totalTarBulanIni++; } } }); // Batasan Target Maksimal Pelatihan const targetDarDaily = 10; const targetDarMonthly = 10 * jumlahHariBulanIni; const targetTarMonthly = 2; // Hitung Persentase Akhir let pDarDaily = Math.min(Math.round((totalJamHariIni / targetDarDaily) * 100), 100); let pDarMonthly = Math.min(Math.round((totalJamBulanIni / targetDarMonthly) * 100), 100); let pTarMonthly = Math.min(Math.round((totalTarBulanIni / targetTarMonthly) * 100), 100); let pSummary = Math.round((pDarDaily + pDarMonthly + pTarMonthly) / 3); // Perbarui Komponen Tampilan Visual Lingkaran updateCircularProgress(pDarDaily, pDarMonthly, pTarMonthly, pSummary, totalJamHariIni, totalJamBulanIni, targetDarMonthly, totalTarBulanIni); }
