function hitungKalkulasiProgress(rowsDARSiswa, rowsTARSiswa) { const hariIni = new Date(); const tahunSekarang = hariIni.getFullYear(); const bulanSekarang = hariIni.getMonth(); const tglSekarang = hariIni.getDate(); const namaBulan = ["Januari", "Februari", "Maret", "April", "Mei", "Juni", "Juli", "Agustus", "September", "Oktober", "November", "Desember"]; document.getElementById('title-kalender').innerText = `📆 KELENGKAPAN DAR HARIAN (${namaBulan[bulanSekarang].toUpperCase()} ${tahunSekarang})`; const jumlahHariBulanIni = new Date(tahunSekarang, bulanSekarang + 1, 0).getDate(); let mapJamPerHari = {}; for(let d = 1; d <= jumlahHariBulanIni; d++) { mapJamPerHari[d] = 0; } let totalJamHariIni = 0; let totalJamBulanIni = 0; let totalTarBulanIni = 0; // 1. FUNGSI PARSING TANGGAL function parsingTanggalGSheets(tglRaw) { if (!tglRaw) return null; if (tglRaw instanceof Date && !isNaN(tglRaw)) { if (tglRaw.getFullYear() === 1899) return null; return { hari: tglRaw.getDate(), bulan: tglRaw.getMonth(), tahun: tglRaw.getFullYear() }; } let tglStr = tglRaw.toString().trim(); if (tglStr === "" || tglStr.toLowerCase().includes("null") || tglStr.toLowerCase().includes("undefined") || tglStr.includes("1899")) return null; if (tglStr.includes('-')) { const bagianKiri = tglStr.split('T')[0]; const p = bagianKiri.split('-'); if (p.length === 3) { let thn = parseInt(p[0]); let bln = parseInt(p[1]) - 1; let tgl = parseInt(p[2]); if (thn > 1900) return { hari: tgl, bulan: bln, tahun: thn }; } } if (tglStr.includes('/')) { const p = tglStr.split('/'); if (p.length === 3) { if (p[0].length === 4) { return { hari: parseInt(p[2]), bulan: parseInt(p[1]) - 1, tahun: parseInt(p[0]) }; } else { return { hari: parseInt(p[0]), bulan: parseInt(p[1]) - 1, tahun: parseInt(p[2]) }; } } } let ujiDate = new Date(tglStr); if (!isNaN(ujiDate) && ujiDate.getFullYear() > 1900) { return { hari: ujiDate.getDate(), bulan: ujiDate.getMonth(), tahun: ujiDate.getFullYear() }; } return null; } // 2. FUNGSI BARU: PARSING DURASI JAM SECARA AKURAT function parsingDurasiGSheets(durasiRaw) { if (durasiRaw === null || durasiRaw === undefined) return 0; // Jika berupa objek Date bawaan GSheets ({date: "1899-12-30T11:00:00.000Z"}) if (typeof durasiRaw === 'object' && durasiRaw.date) { let d = new Date(durasiRaw.date); return d.getHours() + (d.getMinutes() / 60); } let str = durasiRaw.toString().trim(); if (str === "") return 0; // Jika string mengandung format tahun berekor 1899 if (str.includes('1899')) { let d = new Date(str); if (!isNaN(d)) { return d.getHours() + (d.getMinutes() / 60); } } // Jika berbentuk format waktu string biasa seperti "11:00" atau "11.00" if (str.includes(':') || str.includes('.')) { const pemisah = str.includes(':') ? ':' : '.'; const bagian = str.split(pemisah); const jam = parseFloat(bagian[0]) || 0; const menit = parseFloat(bagian[1]) || 0; if (jam < 24) { return jam + (menit / 60); } } // Jika sudah berbentuk angka desimal murni murni biasa (contoh: 11 atau 10.5) let angkaMurni = parseFloat(str); if (!isNaN(angkaMurni)) { return angkaMurni; } return 0; } // LOOP DATA DAR SISWA rowsDARSiswa.forEach(row => { // Kunci murni mencari nama kolom yang mengandung kata 'durasi' (Aman dari jebakan kata 'jam') const keyDurasi = Object.keys(row).find(k => k.toLowerCase().includes('durasi')); const keyTanggal = Object.keys(row).find(k => k.toLowerCase().includes('tanggal')); let durasi = 0; if (keyDurasi) { durasi = parsingDurasiGSheets(row[keyDurasi]); } let tglRaw = row[keyTanggal]; const dataTgl = parsingTanggalGSheets(tglRaw); if (dataTgl) { if (dataTgl.bulan === bulanSekarang && dataTgl.tahun === tahunSekarang) { totalJamBulanIni += durasi; if(mapJamPerHari[dataTgl.hari] !== undefined) { mapJamPerHari[dataTgl.hari] += durasi; } if (dataTgl.hari === tglSekarang) { totalJamHariIni += durasi; } } } }); // LOOP DATA TAR SISWA rowsTARSiswa.forEach(row => { const keyTanggal = Object.keys(row).find(k => k.toLowerCase().includes('tanggal') || k.toLowerCase().includes('analisis')); let tglRaw = row[keyTanggal]; const dataTgl = parsingTanggalGSheets(tglRaw); if (dataTgl) { if (dataTgl.bulan === bulanSekarang && dataTgl.tahun === tahunSekarang) { totalTarBulanIni++; } } }); // PENGHITUNGAN TARGET SESUAI INSTRUKSI BARU const targetDarDaily = 10; const targetDarMonthly = 10 * jumlahHariBulanIni; const targetTarMonthly = 2; let pDarDaily = Math.min(Math.round((totalJamHariIni / targetDarDaily) * 100), 100); let pDarMonthly = Math.min(Math.round((totalJamBulanIni / targetDarMonthly) * 100), 100); let pTarMonthly = Math.min(Math.round((totalTarBulanIni / targetTarMonthly) * 100), 100); updateCircularProgress(pDarDaily, pDarMonthly, pTarMonthly, totalJamHariIni, totalJamBulanIni, targetDarMonthly, totalTarBulanIni); renderKalenderDAR(jumlahHariBulanIni, mapJamPerHari); }
